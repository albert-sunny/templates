		\question
		There is a communication channel of maximum capacity $1 \, unit$. There are $n$ users who simultaneously use this channel. User $i$ wishes to send $x_i \in [0,1]$ units of data on this channel. The payoff of user $i$ has the following form
		$$u_i(x_i,x_{-i}) = \begin{cases}
		1 + x_i(a - \sum^n_{j=1} x_j) & \textrm{ if } \sum^n_{j=1} x_j \leq a \\
		0  & \textrm{ otherwise}
		\end{cases} $$
		where $a \in [0,1]$ is fixed by the network operator to cap the amount of data transmitted by the users.
		
		\begin{parts}
			\part[10]
			Does the above problem have a \emph{pure strategy Nash equilibrium (PSNE)} for all values of $a \in [0,1]$? Compute the \textit{PSNE} when it exists. \droppoints
			\part[4]
			Does the above problem have a \emph{dominant strategy equilibrium} for any value of $a$? Provide proper steps/arguments to justify your answer. \droppoints
		\end{parts}	

		
		\question
		Given an example for each of the following
		\begin{parts}
			\part[2]
			A 2-player game with \emph{\textbf{one} dominant strategy equilibrium} and \emph{\textbf{no} pure strategy Nash equilibrium}. \droppoints
			
			\part[2]
			A 2-player game with \emph{\textbf{two} pure strategy Nash equilibrium} and \emph{\textbf{no} dominant strategy equilibrium}. \droppoints
			
			\part[2]
			A 2-player game with \emph{\textbf{at-least} one mixed strategy Nash equilibrium} and \emph{\textbf{no} pure strategy Nash equilibrium}. \droppoints
		\end{parts}

	\end{questions}
